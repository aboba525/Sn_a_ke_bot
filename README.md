  <h1>üêç –§—Ä—É–∫—Ç–æ–≤–∞—è –∑–º–µ–π–∫–∞</h1>

  <script>
    // Telegram WebApp
    const tg = window.Telegram?.WebApp;
    if (tg) {
      tg.ready();
      tg.expand();
    }

    // –≠–ª–µ–º–µ–Ω—Ç—ã
    const canvas = document.getElementById("game");
    const ctx = canvas.getContext("2d");
    const scoreDisplay = document.getElementById("score");
    const bestDisplay = document.getElementById("best");

    // –ù–∞—Å—Ç—Ä–æ–π–∫–∏
    const gridSize = 15;
    const tileCount = 20;

    // –§—Ä—É–∫—Ç—ã
    const FRUITS = [
      { emoji: "üçé", points: 10, weight: 6 },
      { emoji: "üçå", points: 10, weight: 5 },
      { emoji: "ü•ù", points: 10, weight: 5 },
      { emoji: "üçí", points: 10, weight: 4 },
      { emoji: "üçà", points: 20, weight: 2, big: true },
      { emoji: "üçâ", points: 20, weight: 2, big: true }
    ];

    // –°–æ—Å—Ç–æ—è–Ω–∏–µ
    let snake = [{ x: 10, y: 10 }];
    let dx = 0, dy = 0;
    let score = 0;
    let bestScore = parseInt(localStorage.getItem("snakeBest")) || 0;
    let food = null;
    let gameRunning = false;
    let gameStarted = false;
    let gameInterval;

    // –ó–≤—É–∫–∏ (–∫–æ—Ä–æ—Ç–∫–∏–µ base64)
    const eatSound = new Audio("data:audio/wav;base64,UklGRiQAAABXQVZFZm10IBAAAAABAAEARKwAAIhYAQACABAAZGF0YQAAAAA=");
    const dieSound = new Audio("data:audio/wav;base64,UklGRiQAAABXQVZFZm10IBAAAAABAAEARKwAAIhYAQACABAAZGF0YQAAAAI=");
    const startSound = new Audio("data:audio/wav;base64,UklGRiQAAABXQVZFZm10IBAAAAABAAEARKwAAIhYAQACABAAZGF0YQAAABA=");

    eatSound.volume = 0.3;
    dieSound.volume = 0.4;
    startSound.volume = 0.5;

    // –†–µ–∫–æ—Ä–¥
    bestDisplay.textContent = `–†–µ–∫–æ—Ä–¥: ${bestScore}`;

    // –í—ã–±–æ—Ä —Ñ—Ä—É–∫—Ç–∞ —Å —É—á—ë—Ç–æ–º –≤–µ—Å–∞
    function getRandomFruit() {
      const totalWeight = FRUITS.reduce((sum, f) => sum + f.weight, 0);
      let rand = Math.random() * totalWeight;
      for (let fruit of FRUITS) {
        rand -= fruit.weight;
        if (rand <= 0) return { ...fruit };
      }
      return FRUITS[0];
    }

    // –ê–Ω–∏–º–∞—Ü–∏—è —Å—Ç–∞—Ä—Ç–∞
    function showStartAnimation() {
      ctx.fillStyle = "black";
      ctx.fillRect(0, 0, canvas.width, canvas.height);
      
      let step = 0;
      const anim = setInterval(() => {
        ctx.fillStyle = "rgba(0,0,0,0.8)";
        ctx.fillRect(0, 0, canvas.width, canvas.height);
        
        ctx.fillStyle = "lime";
        ctx.font = "20px Arial";
        ctx.fillText("üêç –ó–ú–ï–ô–ö–ê", 90, 120);
        
        ctx.fillStyle = "white";
        ctx.font = "16px Arial";
        const texts = ["–ø—Ä–æ—Å—ã–ø–∞–µ—Ç—Å—è...", "—Ç—è–Ω–µ—Ç—Å—è...", "–≥–æ–ª–æ–¥–Ω–∞!"];
        ctx.fillText(texts[step % 3], 95, 150);
        
        step++;
        if (step > 15) {
          clearInterval(anim);
          draw();
          showHint();
        }
      }, 200);
    }

    function showHint() {
      ctx.fillStyle = "rgba(0, 0, 0, 0.5)";
      ctx.fillRect(0, 0, canvas.width, canvas.height);
      ctx.fillStyle = "white";
      ctx.font = "18px Arial";
      ctx.fillText("–ù–∞–∂–º–∏ —Å—Ç—Ä–µ–ª–∫—É", 90, 140);
      ctx.fillText("—á—Ç–æ–±—ã –Ω–∞—á–∞—Ç—å", 90, 170);
    }

    function placeFood() {
      food = getRandomFruit();
      food.x = Math.floor(Math.random() * tileCount);
      food.y = Math.floor(Math.random() * tileCount);
      
      for (let part of snake) {
        if (part.x === food.x && part.y === food.y) {
          placeFood();
        }
      }
    }

    function draw() {
      ctx.fillStyle = "black";
      ctx.fillRect(0, 0, canvas.width, canvas.height);

      // –ó–º–µ–π–∫–∞
      ctx.fillStyle = "lime";
      for (let part of snake) {
        ctx.fillRect(part.x * gridSize, part.y * gridSize, gridSize - 2, gridSize - 2);
      }

      // –§—Ä—É–∫—Ç
      if (food) {
        ctx.font = food.big ? "24px Arial" : "18px Arial";
        ctx.fillText(food.emoji, food.x * gridSize + 2, food.y * gridSize + 14);
      }
    }

    function setDir(x, y) {
      if (!gameStarted) {
        gameStarted = true;
        gameRunning = true;
        startSound.play();
        clearInterval(gameInterval);
        gameInterval = setInterval(gameLoop, 150);
      }

      if (x !== 0 && dx === 0) { dx = x; dy = 0; }
      if (y !== 0 && dy === 0) { dy = y; dx = 0; }
    }

    function gameLoop() {
      if (!gameRunning) return;

      const head = { x: snake[0].x + dx, y: snake[0].y + dy };

      if (head.x < 0 || head.x >= tileCount || head.y < 0 || head.y >= tileCount) {
        gameOver();
        return;
      }

      for (let part of snake) {
        if (part.x === head.x && part.y === head.y) {
          gameOver();
          return;
        }
      }

      snake.unshift(head);

      if (head.x === food.x && head.y === food.y) {
        score += food.points;
        scoreDisplay.textContent = `–û—á–∫–∏: ${score}`;
        eatSound.play();
        placeFood();
      } else {
        snake.pop();
      }

      draw();
    }

    function gameOver() {
      gameRunning = false;
      clearInterval(gameInterval);
      dieSound.play();

      // –û–±–Ω–æ–≤–ª—è–µ–º —Ä–µ–∫–æ—Ä–¥
      if (score > bestScore) {
        bestScore = score;
        localStorage.setItem("snakeBest", bestScore);
        bestDisplay.textContent = `–†–µ–∫–æ—Ä–¥: ${bestScore} üåü`;
      }

      ctx.fillStyle = "rgba(0,0,0,0.8)";
      ctx.fillRect(0, 0, canvas.width, canvas.height);
      ctx.fillStyle = "red";
      ctx.font = "30px Arial";
      ctx.fillText("–ü–û–¢–†–ê–ß–ï–ù–û!", 60, 140);
      ctx.fillStyle = "white";
      ctx.font = "20px Arial";
      ctx.fillText(`–û—á–∫–∏: ${score}`, 100, 180);
    }

    function restart() {
      clearInterval(gameInterval);
      snake = [{ x: 10, y: 10 }];
      dx = 0; dy = 0;
      score = 0;
      scoreDisplay.textContent = "–û—á–∫–∏: 0";
      gameRunning = false;
      gameStarted = false;
      placeFood();
      draw();
      showStartAnimation();
    }

    // –°—Ç–∞—Ä—Ç
    window.onload = () => {
      placeFood();
      draw();
      showStartAnimation();
    };
